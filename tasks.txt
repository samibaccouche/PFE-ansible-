---
- name: Installer les dépendances (ACL et Driver Python)
  apt:
    name: ['acl', 'postgresql', 'python3-psycopg2']
    state: present
    update_cache: yes

- name: Détecter la version de PostgreSQL
  shell: ls /etc/postgresql/ | head -n 1
  register: pg_version_installed
  changed_when: false

- name: Configurer l'écoute sur toutes les interfaces
  lineinfile:
    path: "/etc/postgresql/{{ pg_version_installed.stdout }}/main/postgresql.conf"
    regexp: "^#?listen_addresses ="
    line: "listen_addresses = '*'"
  notify: restart postgresql

- name: 1. Créer les Rôles de Groupe (NOLOGIN)
  become: yes
  become_user: postgres
  community.postgresql.postgresql_user:
    name: "{{ item }}"
    role_attr_flags: "NOLOGIN"
    state: present
  loop:
    - "db_analyst"
    - "db_developer"

- name: 2. Vérifier si les bases existent déjà (SANS modification)
  become: yes
  become_user: postgres
  community.postgresql.postgresql_db:
    name: "{{ item.name }}"
    state: present
  loop: "{{ applications }}"
  register: db_check
  changed_when: false
  ignore_errors: yes

- name: 3. Créer les bases UNIQUEMENT si elles n'existent pas
  become: yes
  become_user: postgres
  community.postgresql.postgresql_db:
    name: "{{ item.item.name }}"
    state: present
  loop: "{{ db_check.results }}"
  when: item.failed or item.db is not defined
  loop_control:
    label: "{{ item.item.name }}"

- name: 4. Copier le fichier de schéma sur la VM2
  copy:
    src: "files/{{ item.item.schema_file }}"
    dest: "/tmp/{{ item.item.schema_file }}"
  loop: "{{ db_check.results }}"
  when: (item.failed or item.db is not defined) and item.item.schema_file is defined
  loop_control:
    label: "{{ item.item.name }}"

- name: 5. Importer le schéma UNIQUEMENT au premier déploiement
  become: yes
  become_user: postgres
  shell: "psql -d {{ item.item.name }} -f /tmp/{{ item.item.schema_file }}"
  loop: "{{ db_check.results }}"
  when: (item.failed or item.db is not defined) and item.item.schema_file is defined
  loop_control:
    label: "{{ item.item.name }}"
  notify: restart postgresql

- name: 6. Créer les utilisateurs (LOGIN)
  become: yes
  become_user: postgres
  community.postgresql.postgresql_user:
    name: "{{ item.user }}"
    password: "{{ item.pass }}"
    role_attr_flags: "LOGIN"
    state: present
  loop:
    - { user: "{{ analyst_user }}", pass: "{{ analyst_password }}" }
    - { user: "{{ developer_user }}", pass: "{{ developer_password }}" }
    - { user: "{{ applications[0].user }}", pass: "{{ applications[0].pass }}" }

- name: 7. Lier les utilisateurs aux groupes (Membership)
  become: yes
  become_user: postgres
  community.postgresql.postgresql_membership:
    group: "{{ item.group }}"
    target_role: "{{ item.user }}"
    state: present
  loop:
    - { user: "{{ analyst_user }}", group: "db_analyst" }
    - { user: "{{ developer_user }}", group: "db_developer" }
    - { user: "{{ applications[0].user }}", group: "db_developer" }

- name: 8. Autoriser l'accès distant (VM1 control-plane)
  become: yes
  become_user: postgres
  community.postgresql.postgresql_pg_hba:
    dest: "/etc/postgresql/{{ pg_version_installed.stdout }}/main/pg_hba.conf"
    contype: host
    databases: "{{ item.name }}"
    users: "{{ item.user }}"
    source: "{{ app_server_ip }}/32"
    method: md5
    create: true
  loop: "{{ applications }}"
  notify: restart postgresql
